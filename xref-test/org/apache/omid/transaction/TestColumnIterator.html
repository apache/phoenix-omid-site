35  @Test(groups = "noHBase")
36  public class TestColumnIterator {
37  
38      final byte[] row = Bytes.toBytes("row");
39      private final byte[] family1 = Bytes.toBytes("f1");
40      private final byte[] family2 = Bytes.toBytes("f2");
41      private final byte[] qualifier1 = Bytes.toBytes("c1");
42      private final byte[] qualifier2 = Bytes.toBytes("c2");
43      final byte[] data = Bytes.toBytes("data");
44  
45      private final List<Cell> cells = new ArrayList<Cell>(
46              Arrays.asList(
47                      // Group 1 (3 elems but grouping should filter shadow cell, so check for 2)
48                      new KeyValue(row, family1, qualifier1, 0, data),
49                      new KeyValue(row, family1, qualifier1, 1, data),
50                      new KeyValue(row, family1, CellUtils.addShadowCellSuffix(qualifier1), 0, data),
51                      // Group 2 (2 elems but grouping should filter shadow cell, so check for 1)
52                      new KeyValue(row, family1, qualifier2, 0, data),
53                      new KeyValue(row, family1, CellUtils.addShadowCellSuffix(qualifier2), 0, data),
54                      // Group 3 (2 elems but grouping should filter shadow cell, so check for 1)
55                      new KeyValue(row, family2, qualifier1, 0, data),
56                      new KeyValue(row, family2, CellUtils.addShadowCellSuffix(qualifier1), 0, data)
57              )
58      );
59  
60      @Test
61      public void testGroupingCellsByColumnFilteringShadowCells() {
62  
63          ImmutableList<Collection<Cell>> groupedColumnsWithoutShadowCells =
64                  TTable.groupCellsByColumnFilteringShadowCells(cells);
65          Log.info("Column Groups " + groupedColumnsWithoutShadowCells);
66          assertEquals("Should be 3 column groups", 3, groupedColumnsWithoutShadowCells.size());
67          int group1Counter = 0;
68          int group2Counter = 0;
69          int group3Counter = 0;
70          for (Collection<Cell> columns : groupedColumnsWithoutShadowCells) {
71              for (Cell cell : columns) {
72                  byte[] cellFamily = CellUtil.cloneFamily(cell);
73                  byte[] cellQualifier = CellUtil.cloneQualifier(cell);
74                  // Group 1
75                  if (Bytes.equals(cellFamily, family1) &&
76                          Bytes.equals(cellQualifier, qualifier1)) {
77                      group1Counter++;
78                  }
79                  // Group 2
80                  if (Bytes.equals(cellFamily, family1) &&
81                          Bytes.equals(cellQualifier, qualifier2)) {
82                      group2Counter++;
83                  }
84                  // Group 3
85                  if (Bytes.equals(cellFamily, family2) &&
86                          Bytes.equals(cellQualifier, qualifier1)) {
87                      group3Counter++;
88                  }
89              }
90          }
91  
92          assertEquals("Group 1 should have 2 elems", 2, group1Counter);
93          assertEquals("Group 2 should have 1 elems", 1, group2Counter);
94          assertEquals("Group 3 should have 1 elems", 1, group3Counter);
95      }
96  }
